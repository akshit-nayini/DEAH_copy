# DEAH v1 — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a complete, production-ready CLI multi-agent system (DEAH v1) that accepts data engineering requirements, validates them, routes to 3 agents, and outputs design blueprints.

**Architecture:** ENV-driven LLM (anthropic/gemini/mock) + Jira integration (real/mock fallback). Three agents (Architecture 2-stage, Data Model, Impl Spec) are orchestrated by a routing layer that reads `activity_type`. A cross-artifact validator + auto-fix loop with human-in-the-loop CLI checkpoints produces 5 output files.

**Tech Stack:** Python 3.11+, Pydantic v2, Typer, Rich, python-dotenv, PyYAML, requests, anthropic SDK, google-generativeai SDK

```python
from __future__ import annotations
from enum import Enum
from typing import Optional
from pydantic import BaseModel, field_validator


class ActivityType(str, Enum):
    new_dev = "new_dev"
    enhancement = "enhancement"
    bug = "bug"


class EnvironmentType(str, Enum):
    DEV = "DEV"
    QA = "QA"
    PROD = "PROD"


class RequirementDoc(BaseModel):
    # ── Core (mandatory) ──────────────────────────────────────────────────
    activity_type: ActivityType
    use_case: str
    acceptance_criteria: list[str]

    # ── Optional ──────────────────────────────────────────────────────────
    title: Optional[str] = None
    technology: Optional[dict[str, str]] = None
    environment: Optional[EnvironmentType] = None
    connections: Optional[dict[str, str]] = None
    business_context: Optional[str] = None
    constraints: Optional[list[str]] = None
    assumptions: Optional[list[str]] = None
    non_functional_requirements: Optional[list[str]] = None
    data_volume: Optional[str] = None
    latency_requirements: Optional[str] = None

    @field_validator("acceptance_criteria")
    @classmethod
    def at_least_one_criterion(cls, v: list[str]) -> list[str]:
        if not v:
            raise ValueError("acceptance_criteria must have at least one entry")
        return v

    def summary(self) -> str:
        lines = [
            f"Activity Type : {self.activity_type.value}",
            f"Use Case      : {self.use_case}",
            f"Criteria      : {', '.join(self.acceptance_criteria[:3])}",
        ]
        if self.title:
            lines.insert(0, f"Title         : {self.title}")
        if self.environment:
            lines.append(f"Environment   : {self.environment.value}")
        if self.technology:
            lines.append(f"Technology    : {self.technology}")
        if self.data_volume:
            lines.append(f"Data Volume   : {self.data_volume}")
        return "\n".join(lines)


def validate_core_fields(data: dict) -> list[str]:
    """Return list of missing core field names (empty = all present)."""
    core = ["activity_type", "use_case", "acceptance_criteria"]
    return [f for f in core if not data.get(f)]
```

```python
from __future__ import annotations
import os
import yaml
from pathlib import Path
from pydantic_settings import BaseSettings, SettingsConfigDict
from pydantic import Field


_CONFIG_PATH = Path(__file__).parent.parent / "config.yaml"


def _load_yaml() -> dict:
    if _CONFIG_PATH.exists():
        with open(_CONFIG_PATH) as f:
            return yaml.safe_load(f) or {}
    return {}


class Settings(BaseSettings):
    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        extra="ignore",
    )

    llm_provider: str = Field(default="mock", alias="LLM_PROVIDER")
    anthropic_api_key: str = Field(default="", alias="ANTHROPIC_API_KEY")
    gemini_api_key: str = Field(default="", alias="GEMINI_API_KEY")
    model_name: str = Field(default="", alias="MODEL_NAME")
    max_retries: int = Field(default=3, alias="MAX_RETRIES")
    confidence_threshold: float = Field(default=0.75, alias="CONFIDENCE_THRESHOLD")
    jira_base_url: str = Field(default="", alias="JIRA_BASE_URL")
    jira_email: str = Field(default="", alias="JIRA_EMAIL")
    jira_api_token: str = Field(default="", alias="JIRA_API_TOKEN")

    @property
    def jira_available(self) -> bool:
        return bool(self.jira_base_url and self.jira_email and self.jira_api_token)

    @property
    def effective_llm_provider(self) -> str:
        """Auto-fallback to mock if API key missing."""
        p = self.llm_provider.lower()
        if p == "anthropic" and not self.anthropic_api_key:
            return "mock"
        if p == "gemini" and not self.gemini_api_key:
            return "mock"
        return p

    @property
    def effective_model_name(self) -> str:
        if self.model_name:
            return self.model_name
        yaml_cfg = _load_yaml()
        defaults = yaml_cfg.get("llm_defaults", {})
        p = self.effective_llm_provider
        if p == "anthropic":
            return defaults.get("anthropic_model", "claude-3-5-sonnet-20241022")
        if p == "gemini":
            return defaults.get("gemini_model", "gemini-1.5-pro")
        return "mock"


class ScoringWeights:
    def __init__(self):
        yaml_cfg = _load_yaml()
        w = yaml_cfg.get("scoring_weights", {})
        self.cost: float = w.get("cost", 0.30)
        self.scalability: float = w.get("scalability", 0.25)
        self.complexity: float = w.get("complexity", 0.20)
        self.latency: float = w.get("latency", 0.15)
        self.operability: float = w.get("operability", 0.10)

    def as_dict(self) -> dict[str, float]:
        return {
            "cost": self.cost,
            "scalability": self.scalability,
            "complexity": self.complexity,
            "latency": self.latency,
            "operability": self.operability,
        }


def get_settings() -> Settings:
    return Settings()


def get_scoring_weights() -> ScoringWeights:
    return ScoringWeights()
```

```python
from __future__ import annotations
import anthropic as _anthropic


class AnthropicClient:
    def __init__(self, api_key: str, model: str):
        self._client = _anthropic.Anthropic(api_key=api_key)
        self._model = model

    def complete(self, prompt: str) -> str:
        msg = self._client.messages.create(
            model=self._model,
            max_tokens=4096,
            messages=[{"role": "user", "content": prompt}],
        )
        return msg.content[0].text
```

```python
from __future__ import annotations
from typing import Protocol, runtime_checkable
from deah.config import get_settings
from deah.llm.mock_client import get_mock_response


@runtime_checkable
class LLMClient(Protocol):
    def complete(self, prompt: str) -> str: ...


class MockClient:
    def complete(self, prompt: str) -> str:
        return get_mock_response(prompt)


def get_llm_client() -> LLMClient:
    """Factory: returns real or mock client based on .env."""
    settings = get_settings()
    provider = settings.effective_llm_provider
    model = settings.effective_model_name

    if provider == "anthropic":
        from deah.llm.anthropic_client import AnthropicClient
        return AnthropicClient(api_key=settings.anthropic_api_key, model=model)

    if provider == "gemini":
        from deah.llm.gemini_client import GeminiClient
        return GeminiClient(api_key=settings.gemini_api_key, model=model)

    return MockClient()
```

```python
from __future__ import annotations
import re
import yaml


def _extract_text_from_adf(node: dict | list | None) -> str:
    """Recursively extract plain text from Atlassian Document Format (ADF)."""
    if node is None:
        return ""
    if isinstance(node, list):
        return "\n".join(_extract_text_from_adf(n) for n in node)
    if isinstance(node, dict):
        if node.get("type") == "text":
            return node.get("text", "")
        children = node.get("content", [])
        return "\n".join(_extract_text_from_adf(c) for c in children)
    return str(node)


def _parse_yaml_block(text: str) -> dict:
    """Parse YAML-formatted requirement fields from free text."""
    try:
        parsed = yaml.safe_load(text)
        if isinstance(parsed, dict):
            return parsed
    except Exception:
        pass
    return {}


def _extract_field(text: str, field: str) -> str | None:
    """Extract single-line field value with regex fallback."""
    pattern = rf"^{re.escape(field)}\s*:\s*(.+)$"
    m = re.search(pattern, text, re.MULTILINE | re.IGNORECASE)
    return m.group(1).strip() if m else None


def _extract_list_field(text: str, field: str) -> list[str]:
    """Extract a YAML list field from text."""
    pattern = rf"^{re.escape(field)}\s*:\s*\n((?:\s+-\s+.+\n?)+)"
    m = re.search(pattern, text, re.MULTILINE | re.IGNORECASE)
    if m:
        items = re.findall(r"^\s+-\s+(.+)$", m.group(1), re.MULTILINE)
        return [i.strip() for i in items]
    # Inline list fallback
    inline = _extract_field(text, field)
    if inline:
        return [i.strip() for i in inline.split(",") if i.strip()]
    return []


def parse_jira_response(issue: dict) -> dict:
    """
    Parse a Jira REST API issue dict into a RequirementDoc-compatible dict.
    Supports both Atlassian Document Format (ADF) and plain text descriptions.
    """
    fields = issue.get("fields", {})
    summary = fields.get("summary", "")

    # Extract description text
    desc_raw = fields.get("description", "")
    if isinstance(desc_raw, dict):
        desc_text = _extract_text_from_adf(desc_raw)
    else:
        desc_text = str(desc_raw or "")

    # Extract comment texts
    comments_raw = fields.get("comment", {}).get("comments", [])
    comment_texts = []
    for c in comments_raw:
        body = c.get("body", "")
        if isinstance(body, dict):
            comment_texts.append(_extract_text_from_adf(body))
        else:
            comment_texts.append(str(body or ""))
    all_text = desc_text + "\n" + "\n".join(comment_texts)

    # Try YAML block parse first (most structured)
    parsed = _parse_yaml_block(all_text) or {}

    result: dict = {}

    # Core fields
    result["activity_type"] = (
        parsed.get("activity_type")
        or _extract_field(all_text, "activity_type")
        or "new_dev"
    )
    result["use_case"] = (
        parsed.get("use_case")
        or _extract_field(all_text, "use_case")
        or desc_text.strip()[:500]
    )
    ac = parsed.get("acceptance_criteria") or _extract_list_field(all_text, "acceptance_criteria")
    result["acceptance_criteria"] = ac if isinstance(ac, list) else ([ac] if ac else [])

    # Optional fields
    result["title"] = summary or parsed.get("title")
    for opt in ["environment", "data_volume", "latency_requirements", "business_context"]:
        val = parsed.get(opt) or _extract_field(all_text, opt)
        if val:
            result[opt] = val

    for list_opt in ["constraints", "assumptions", "non_functional_requirements"]:
        val = parsed.get(list_opt) or _extract_list_field(all_text, list_opt)
        if val:
            result[list_opt] = val if isinstance(val, list) else [val]

    tech = parsed.get("technology")
    if tech and isinstance(tech, dict):
        result["technology"] = tech

    conns = parsed.get("connections")
    if conns and isinstance(conns, dict):
        result["connections"] = conns

    return result
```

```python
            result.stage1_output = self._llm.complete(stage1_prompt)

                stage1_output=result.stage1_output,
                weights=self._weights.as_dict(),

            result.full_output = (
                "# Architecture Options\n\n"
                + result.stage1_output
                + "\n\n---\n\n# Scoring & Recommendation\n\n"
                + result.stage2_output
            )
        except Exception as e:
            result.error = str(e)
```

```python
from __future__ import annotations
from deah.models.requirement_doc import RequirementDoc

AGENT_ARCHITECTURE = "architecture"
AGENT_DATA_MODEL = "data_model"
AGENT_IMPL_SPEC = "impl_spec"

# Keywords that signal schema/data impact regardless of activity_type
_SCHEMA_IMPACT_KEYWORDS = [
    "schema", "data model", "table", "column", "migration", "scd",
    "dimension", "fact table", "ddl", "alter table", "new column",
    "partition", "clustering", "restructure", "redesign",
]


def _has_schema_impact(req: RequirementDoc) -> bool:
    """Return True if the requirement signals schema or data model changes."""
    text = " ".join([
        req.use_case,
        req.business_context or "",
        " ".join(req.acceptance_criteria),
        " ".join(req.constraints or []),
    ]).lower()
    return any(kw in text for kw in _SCHEMA_IMPACT_KEYWORDS)


def route(req: RequirementDoc) -> list[str]:
    """
    Return ordered list of agent names to run for this requirement.

    Default routing:
      new_dev     → architecture + data_model + impl_spec
      enhancement → data_model + impl_spec
      bug         → impl_spec only

    Override: if schema/data impact detected → add architecture + data_model
    """
    activity = req.activity_type.value
    schema_impact = _has_schema_impact(req)

    if activity == "new_dev":
        return [AGENT_ARCHITECTURE, AGENT_DATA_MODEL, AGENT_IMPL_SPEC]

    if activity == "enhancement":
        if schema_impact:
            return [AGENT_ARCHITECTURE, AGENT_DATA_MODEL, AGENT_IMPL_SPEC]
        return [AGENT_DATA_MODEL, AGENT_IMPL_SPEC]

    # bug
    if schema_impact:
        return [AGENT_ARCHITECTURE, AGENT_DATA_MODEL, AGENT_IMPL_SPEC]
    return [AGENT_IMPL_SPEC]
```

```python
from __future__ import annotations
from dataclasses import dataclass, field
from enum import Enum


class Severity(str, Enum):
    HIGH = "High"
    MEDIUM = "Medium"
    LOW = "Low"


@dataclass
class ValidationIssue:
    severity: Severity
    component: str
    description: str


@dataclass
class ValidationResult:
    passed: bool
    confidence: float
    issues: list[ValidationIssue] = field(default_factory=list)

    def summary(self) -> str:
        status = "PASS" if self.passed else "FAIL"
        lines = [
            f"Status     : {status}",
            f"Confidence : {self.confidence:.2f}",
            f"Issues     : {len(self.issues)}",
        ]
        for issue in self.issues:
            lines.append(f"  [{issue.severity.value}] {issue.component}: {issue.description}")
        return "\n".join(lines)


def _check_architecture(arch: str) -> list[ValidationIssue]:
    issues = []
    if not arch:
        issues.append(ValidationIssue(Severity.HIGH, "Architecture", "Architecture artifact is empty"))
        return issues
    if "recommendation" not in arch.lower() and "recommended" not in arch.lower():
        issues.append(ValidationIssue(Severity.HIGH, "Architecture", "No recommendation found in architecture output"))
    if "option" not in arch.lower():
        issues.append(ValidationIssue(Severity.MEDIUM, "Architecture", "No architecture options found"))
    if "stack" not in arch.lower() and "tech" not in arch.lower():
        issues.append(ValidationIssue(Severity.MEDIUM, "Architecture", "Tech stack not described"))
    return issues


def _check_data_model(model: str) -> list[ValidationIssue]:
    issues = []
    if not model:
        issues.append(ValidationIssue(Severity.HIGH, "Data Model", "Data model artifact is empty"))
        return issues
    if "create table" not in model.lower():
        issues.append(ValidationIssue(Severity.HIGH, "Data Model", "No DDL CREATE TABLE statements found"))
    if "primary key" not in model.lower() and "pk" not in model.lower():
        issues.append(ValidationIssue(Severity.MEDIUM, "Data Model", "No primary key defined in DDL"))
    if "scd" not in model.lower() and "slowly changing" not in model.lower():
        issues.append(ValidationIssue(Severity.LOW, "Data Model", "SCD strategy not mentioned"))
    return issues


def _check_impl_spec(spec: str) -> list[ValidationIssue]:
    issues = []
    if not spec:
        issues.append(ValidationIssue(Severity.HIGH, "Impl Spec", "Implementation spec artifact is empty"))
        return issues
    if "rollback" not in spec.lower():
        issues.append(ValidationIssue(Severity.MEDIUM, "Impl Spec", "Rollback strategy not described"))
    if "dev" not in spec.lower() or "qa" not in spec.lower():
        issues.append(ValidationIssue(Severity.MEDIUM, "Impl Spec", "Environment deployment order incomplete (missing DEV/QA)"))
    if "error" not in spec.lower() and "retry" not in spec.lower():
        issues.append(ValidationIssue(Severity.LOW, "Impl Spec", "Error handling patterns not described"))
    return issues


def validate_artifacts(artifacts: dict[str, str]) -> ValidationResult:
    """
    Cross-validate all artifacts for completeness and consistency.
    artifacts keys: 'architecture', 'data_model', 'impl_spec'
    """
    issues: list[ValidationIssue] = []

    arch = artifacts.get("architecture", "")
    model = artifacts.get("data_model", "")
    spec = artifacts.get("impl_spec", "")

    issues.extend(_check_architecture(arch))
    issues.extend(_check_data_model(model))
    issues.extend(_check_impl_spec(spec))

    # Confidence scoring: penalise by severity
    penalty = sum(
        0.15 if i.severity == Severity.HIGH else
        0.07 if i.severity == Severity.MEDIUM else
        0.03
        for i in issues
    )
    confidence = max(0.0, min(1.0, 1.0 - penalty))

    # Passed if no HIGH severity issues and confidence >= 0.5
    high_issues = [i for i in issues if i.severity == Severity.HIGH]
    passed = len(high_issues) == 0 and confidence >= 0.5

    return ValidationResult(passed=passed, confidence=confidence, issues=issues)
```

```python
from __future__ import annotations
from dataclasses import dataclass, field
from deah.models.requirement_doc import RequirementDoc
from deah.validator import ValidationResult, ValidationIssue, Severity
from deah.llm.provider import LLMClient


@dataclass
class FixAttempt:
    attempt_number: int
    agent_name: str
    issue_summary: str
    result_summary: str
    success: bool


@dataclass
class AutoFixResult:
    attempts: list[FixAttempt] = field(default_factory=list)
    final_artifacts: dict[str, str] = field(default_factory=dict)
    final_validation: ValidationResult | None = None
    total_retries: int = 0


class AutoFixer:
    """Re-runs only the failing agent with validation feedback as context."""

    def __init__(self, llm: LLMClient, max_retries: int = 3):
        self._llm = llm
        self._max_retries = max_retries

    def run(
        self,
        req: RequirementDoc,
        artifacts: dict[str, str],
        validation: ValidationResult,
        validate_fn,
        agent_runners: dict[str, callable],
    ) -> AutoFixResult:
        """
        artifacts: dict with keys 'architecture', 'data_model', 'impl_spec'
        agent_runners: dict mapping agent name → callable(req) → artifact str
        validate_fn: callable(artifacts) → ValidationResult
        """
        result = AutoFixResult(final_artifacts=dict(artifacts), final_validation=validation)

        for attempt in range(1, self._max_retries + 1):
            if validation.passed:
                break

            # Identify agents to fix based on issue components
            agents_to_fix = self._identify_failing_agents(validation.issues, artifacts)
            if not agents_to_fix:
                break

            for agent_name in agents_to_fix:
                if agent_name not in agent_runners:
                    continue

                issue_summary = self._summarise_issues(validation.issues, agent_name)
                try:
                    new_artifact = agent_runners[agent_name](req, feedback=issue_summary)
                    result.final_artifacts[agent_name] = new_artifact
                    success = True
                    result_summary = f"Re-ran {agent_name}, artifact updated"
                except Exception as e:
                    success = False
                    result_summary = f"Re-run failed: {e}"

                result.attempts.append(FixAttempt(
                    attempt_number=attempt,
                    agent_name=agent_name,
                    issue_summary=issue_summary,
                    result_summary=result_summary,
                    success=success,
                ))

            # Re-validate
            validation = validate_fn(result.final_artifacts)
            result.final_validation = validation
            result.total_retries = attempt

        return result

    @staticmethod
    def _identify_failing_agents(
        issues: list[ValidationIssue], artifacts: dict[str, str]
    ) -> list[str]:
        """Map HIGH/MEDIUM issues to their responsible agents."""
        component_to_agent = {
            "architecture": "architecture",
            "data model": "data_model",
            "impl spec": "impl_spec",
        }
        agents = []
        for issue in issues:
            if issue.severity in (Severity.HIGH, Severity.MEDIUM):
                comp = issue.component.lower()
                for keyword, agent in component_to_agent.items():
                    if keyword in comp and agent not in agents and agent in artifacts:
                        agents.append(agent)
        return agents

    @staticmethod
    def _summarise_issues(issues: list[ValidationIssue], agent_name: str) -> str:
        component_map = {
            "architecture": "architecture",
            "data_model": "data model",
            "impl_spec": "impl spec",
        }
        component_filter = component_map.get(agent_name, agent_name)
        relevant = [
            f"[{i.severity.value}] {i.description}"
            for i in issues
            if component_filter in i.component.lower()
        ]
        return "; ".join(relevant) if relevant else "General quality improvement needed"
```

```python
from __future__ import annotations
import logging
from dataclasses import dataclass, field
from pathlib import Path

from deah.models.requirement_doc import RequirementDoc, validate_core_fields
from deah.config import get_settings
from deah.llm.provider import get_llm_client
from deah.jira.client import get_jira_client, fetch_requirement_from_jira
from deah.jira.parser import parse_jira_response
from deah.routing import route, AGENT_ARCHITECTURE, AGENT_DATA_MODEL, AGENT_IMPL_SPEC
from deah.agents.architecture import ArchitectureAgent
from deah.agents.data_model import DataModelAgent
from deah.agents.impl_spec import ImplSpecAgent
from deah.validator import validate_artifacts, ValidationResult
from deah.autofix import AutoFixer, FixAttempt
from deah.output import write_all

logger = logging.getLogger(__name__)


@dataclass
class RunResult:
    success: bool
    req: RequirementDoc | None = None
    artifacts: dict[str, str] = field(default_factory=dict)
    validation: ValidationResult | None = None
    output_paths: dict[str, Path] = field(default_factory=dict)
    agents_run: list[str] = field(default_factory=list)
    total_retries: int = 0
    fix_history: list[FixAttempt] = field(default_factory=list)
    error: str = ""
    llm_provider: str = ""
    jira_mode: str = ""


class Orchestrator:
    def __init__(self):
        self._settings = get_settings()
        self._llm = get_llm_client()

    @property
    def llm_mode_label(self) -> str:
        return self._settings.effective_llm_provider.upper()

    @property
    def jira_mode_label(self) -> str:
        return "REAL" if self._settings.jira_available else "MOCK"

    def load_from_markdown(self, path: str) -> dict:
        """Parse a markdown file into a RequirementDoc-compatible dict."""
        import yaml, re
        content = Path(path).read_text(encoding="utf-8")

        # Try YAML front matter first
        fm_match = re.match(r"^---\n(.*?)\n---", content, re.DOTALL)
        if fm_match:
            try:
                parsed = yaml.safe_load(fm_match.group(1))
                if isinstance(parsed, dict):
                    return parsed
            except Exception:
                pass

        # Fallback: parse key: value from markdown body
        result = {}
        lines = content.splitlines()
        current_key = None
        list_items = []

        for line in lines:
            stripped = line.strip()
            if not stripped or stripped.startswith("#"):
                if current_key and list_items:
                    result[current_key] = list_items
                    current_key, list_items = None, []
                continue

            kv_match = re.match(r"^(\w[\w_]*)\s*:\s*(.*)$", stripped)
            if kv_match:
                if current_key and list_items:
                    result[current_key] = list_items
                current_key = kv_match.group(1)
                val = kv_match.group(2).strip()
                list_items = []
                if val:
                    result[current_key] = val
                    current_key = None
            elif stripped.startswith("- ") and current_key:
                list_items.append(stripped[2:].strip())
            elif current_key and not stripped.startswith("-"):
                if current_key not in result:
                    result[current_key] = stripped
                current_key = None

        if current_key and list_items:
            result[current_key] = list_items

        return result

    def load_requirement(self, input_path: str | None, jira_ticket: str | None) -> tuple[RequirementDoc | None, str]:
        """Load and validate RequirementDoc from file or Jira. Returns (doc, error)."""
        if jira_ticket:
            raw = fetch_requirement_from_jira(jira_ticket)
        elif input_path:
            raw = self.load_from_markdown(input_path)
        else:
            return None, "No input provided. Use --input or --jira."

        missing = validate_core_fields(raw)
        if missing:
            return None, f"Missing required core fields: {', '.join(missing)}"

        try:
            doc = RequirementDoc(**raw)
            return doc, ""
        except Exception as e:
            return None, f"Requirement validation error: {e}"

    def run(
        self,
        req: RequirementDoc,
        on_checkpoint1=None,
        on_checkpoint2=None,
    ) -> RunResult:
        """
        Execute full agent pipeline.
        on_checkpoint1: callable(req) → bool (True = proceed, False = abort)
        on_checkpoint2: callable(validation) → str ("approve"|"retry"|"abort")
        """
        result = RunResult(
            success=False,
            req=req,
            llm_provider=self.llm_mode_label,
            jira_mode=self.jira_mode_label,
        )

        settings = self._settings
        agents_to_run = route(req)
        result.agents_run = agents_to_run

        arch_agent = ArchitectureAgent(self._llm)
        dm_agent = DataModelAgent(self._llm)
        is_agent = ImplSpecAgent(self._llm)

        artifacts: dict[str, str] = {}
        arch_summary = ""
        model_summary = ""

        # ── Run agents ────────────────────────────────────────────────────
        if AGENT_ARCHITECTURE in agents_to_run:
            logger.info("Running Architecture Agent (2-stage)...")
            arch_result = arch_agent.run(req)
            if not arch_result.success:
                result.error = f"Architecture Agent failed: {arch_result.error}"
                return result
            artifacts["architecture"] = arch_result.full_output
            arch_summary = arch_result.stage2_output[:500]

        if AGENT_DATA_MODEL in agents_to_run:
            logger.info("Running Data Model Agent...")
            dm_result = dm_agent.run(req)
            if not dm_result.success:
                result.error = f"Data Model Agent failed: {dm_result.error}"
                return result
            artifacts["data_model"] = dm_result.output
            model_summary = dm_result.sql_ddl[:300]

        if AGENT_IMPL_SPEC in agents_to_run:
            logger.info("Running Implementation Spec Agent...")
            is_result = is_agent.run(
                req,
                architecture_summary=arch_summary,
                data_model_summary=model_summary,
            )
            if not is_result.success:
                result.error = f"Impl Spec Agent failed: {is_result.error}"
                return result
            artifacts["impl_spec"] = is_result.output

        result.artifacts = artifacts

        # ── Validate ──────────────────────────────────────────────────────
        logger.info("Validating artifacts...")
        validation = validate_artifacts(artifacts)
        result.validation = validation

        # ── Auto-fix loop ─────────────────────────────────────────────────
        if not validation.passed:
            logger.info(f"Validation failed (confidence={validation.confidence:.2f}). Running auto-fix...")

            def _arch_runner(r: RequirementDoc, feedback: str = "") -> str:
                ar = arch_agent.run(r)
                return ar.full_output

            def _dm_runner(r: RequirementDoc, feedback: str = "") -> str:
                dr = dm_agent.run(r)
                return dr.output

            def _is_runner(r: RequirementDoc, feedback: str = "") -> str:
                ir = is_agent.run(r)
                return ir.output

            fixer = AutoFixer(self._llm, max_retries=settings.max_retries)
            fix_result = fixer.run(
                req=req,
                artifacts=artifacts,
                validation=validation,
                validate_fn=validate_artifacts,
                agent_runners={
                    AGENT_ARCHITECTURE: _arch_runner,
                    AGENT_DATA_MODEL: _dm_runner,
                    AGENT_IMPL_SPEC: _is_runner,
                },
            )
            result.artifacts = fix_result.final_artifacts
            result.validation = fix_result.final_validation
            result.total_retries = fix_result.total_retries
            result.fix_history = fix_result.attempts

        # ── Checkpoint 2: human review if needed ──────────────────────────
        conf_threshold = settings.confidence_threshold
        if on_checkpoint2 and (
            not result.validation.passed
            or result.validation.confidence < conf_threshold
        ):
            decision = on_checkpoint2(result.validation)
            if decision == "abort":
                result.error = "Aborted by user at checkpoint 2"
                return result
            if decision == "retry":
                # Re-run all agents once more
                logger.info("User requested retry — re-running pipeline...")
                return self.run(req, on_checkpoint1=None, on_checkpoint2=on_checkpoint2)

        # ── Write outputs ──────────────────────────────────────────────────
        logger.info("Writing output files...")
        fix_history_dicts = [
            {
                "attempt_number": a.attempt_number,
                "agent_name": a.agent_name,
                "issue_summary": a.issue_summary,
                "result_summary": a.result_summary,
                "success": a.success,
            }
            for a in result.fix_history
        ]
        result.output_paths = write_all(
            artifacts=result.artifacts,
            validation=result.validation,
            request_summary=req.summary(),
            llm_provider=result.llm_provider,
            jira_mode=result.jira_mode,
            retry_count=result.total_retries,
            fix_history=fix_history_dicts,
            request_type=req.activity_type.value,
            agents_run=result.agents_run,
        )

        result.success = True
```

```python
#!/usr/bin/env python3
"""
DEAH v1 — Data Engineering Agent Hub CLI
"""
from __future__ import annotations
import sys
import logging
from pathlib import Path

import typer
from rich.console import Console
from rich.panel import Panel
from rich.table import Table
from rich import print as rprint

from deah.config import get_settings
from deah.orchestrator import Orchestrator
from deah.validator import ValidationResult

app = typer.Typer(
    name="deah",
    help="Data Engineering Agent Hub v1 — Design Blueprint Generator",
    add_completion=False,
)
console = Console()

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    datefmt="%H:%M:%S",
)


def _startup_banner(orch: Orchestrator):
    settings = get_settings()
    console.print(Panel(
        f"[bold cyan]Data Engineering Agent Hub[/bold cyan]  [dim]v1[/dim]\n"
        f"[LLM: [green]{orch.llm_mode_label}[/green]]  "
        f"[JIRA: [yellow]{orch.jira_mode_label}[/yellow]]  "
        f"[Confidence threshold: {settings.confidence_threshold}]  "
        f"[Max retries: {settings.max_retries}]",
        title="DEAH",
        border_style="cyan",
    ))


def _checkpoint1(req) -> bool:
    """Show RequirementDoc summary. Return True to proceed."""
    console.print(Panel(req.summary(), title="[bold]Checkpoint 1 — Requirement Summary[/bold]", border_style="yellow"))
    choice = typer.prompt("Proceed? [A]pprove / [Ab]ort", default="A").strip().lower()
    if choice in ("a", "approve"):
        return True
    console.print("[red]Aborted by user.[/red]")
    return False


def _checkpoint2(validation: ValidationResult) -> str:
    """Show validation summary. Return 'approve', 'retry', or 'abort'."""
    console.print(Panel(
        validation.summary(),
        title="[bold]Checkpoint 2 — Validation Summary[/bold]",
        border_style="red" if not validation.passed else "yellow",
    ))
    choice = typer.prompt(
        "Action? [A]pprove / [R]etry / [Ab]ort", default="A"
    ).strip().lower()
    if choice in ("r", "retry"):
        return "retry"
    if choice in ("ab", "abort"):
        return "abort"
    return "approve"


@app.command()
def run(
    activity: str = typer.Option(
        ..., "--activity", "-a",
        help="Activity type: new_dev | enhancement | bug",
    ),
    input: str = typer.Option(
        None, "--input", "-i",
        help="Path to Markdown requirement file",
    ),
    jira: str = typer.Option(
        None, "--jira", "-j",
        help="Jira ticket ID (e.g. DE-123)",
    ),
    output_dir: str = typer.Option(
        None, "--output-dir", "-o",
        help="Override output directory (default: v1/output/)",
    ),
    no_confirm: bool = typer.Option(
        False, "--no-confirm",
        help="Skip human-in-the-loop checkpoints (non-interactive mode)",
    ),
):
    """
    Run the DEAH pipeline for a data engineering requirement.

    Examples:\n
      python cli.py --activity new_dev --input input_samples/sample.md\n
      python cli.py --activity new_dev --jira DE-123\n
      python cli.py --activity bug --input input_samples/sample.md --no-confirm
    """
    if not input and not jira:
        console.print("[red]Error: provide --input <file> or --jira <ticket-id>[/red]")
        raise typer.Exit(1)

    # Override output dir if requested
    if output_dir:
        import deah.output as out_module
        out_module.OUTPUT_DIR = Path(output_dir)

    orch = Orchestrator()
    _startup_banner(orch)

    # Load requirement
    with console.status("[bold cyan]Loading requirement...[/bold cyan]"):
        req, err = orch.load_requirement(input_path=input, jira_ticket=jira)

    if err:
        console.print(f"[bold red]FAIL: {err}[/bold red]")
        raise typer.Exit(1)

    # Checkpoint 1: human approval of requirement
    if not no_confirm:
        if not _checkpoint1(req):
            raise typer.Exit(0)
    else:
        console.print(Panel(req.summary(), title="Requirement Loaded", border_style="green"))

    # Run pipeline
    with console.status("[bold cyan]Running agents...[/bold cyan]"):
        result = orch.run(
            req,
            on_checkpoint1=None,  # already handled above
            on_checkpoint2=None if no_confirm else _checkpoint2,
        )

    if not result.success:
        console.print(f"[bold red]Pipeline failed: {result.error}[/bold red]")
        raise typer.Exit(1)

    # ── Summary table ──────────────────────────────────────────────────────
    table = Table(title="Run Summary", show_header=True, header_style="bold cyan")
    table.add_column("Field", style="dim")
    table.add_column("Value")

    table.add_row("Activity Type", req.activity_type.value)
    table.add_row("Agents Run", ", ".join(result.agents_run))
    table.add_row("LLM Provider", result.llm_provider)
    table.add_row("Jira Mode", result.jira_mode)
    table.add_row("Retries", str(result.total_retries))
    table.add_row("Confidence", f"{result.validation.confidence:.2f}")
    table.add_row("Validation", "PASS" if result.validation.passed else "FAIL")

    console.print(table)

    console.print("\n[bold green]Output files:[/bold green]")
    for name, path in result.output_paths.items():
        console.print(f"  [cyan]{name}[/cyan]  →  {path}")

    console.print("\n[bold green]Done.[/bold green]")


if __name__ == "__main__":
    app()
```

4. Update `route()` in `deah/routing.py` to include `AGENT_MY_AGENT` for relevant activity types
5. Instantiate and call the agent in `deah/orchestrator.py` alongside existing agents
6. Add output writer in `deah/output.py` if a new file is needed
